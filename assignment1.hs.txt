import Control.Applicative (liftA2)

incrementList :: Functor f => f Int -> f Int
incrementList = fmap (+1)


sumLists :: Applicative f => f Int -> f Int -> f Int
sumLists = liftA2 (+)

main :: IO ()
main = do
    let intList = [1, 2, 3, 4, 5]
        intList2 = [10, 20, 30, 40, 50]
        incremented = incrementList intList
        summed = sumLists intList intList2
    print incremented
    print summed
