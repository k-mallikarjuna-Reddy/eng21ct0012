import Control.Exception (catch)
import System.IO

main :: IO ()
main = do
    contents <- readFile "numbers.txt"
    let numbers = map readInt (lines contents)
        sumResult = sum numbers
    putStrLn $ "Sum: " ++ show sumResult

-- Read an integer, handling non-numeric input
readInt :: String -> Int
readInt s = case reads s of
    [(x, "")] -> x
    _ -> error "Invalid input"
